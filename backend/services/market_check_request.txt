REQUEST:

import axios from 'axios';

const options = {
  method: 'GET',
  url: 'https://api.marketcheck.com/v2/search/car/active',
  params: {
    api_key: 'YOUR_API_KEY',
    year: '2024',
    make: 'ford',
    model: 'f-150',
    start: '0',
    rows: '5',
    facets: 'trim',
    stats: 'price'
  },
  headers: {Accept: 'application/json'}
};

try {
  const { data } = await axios.request(options);
  console.log(data);
} catch (error) {
  console.error(error);
}
